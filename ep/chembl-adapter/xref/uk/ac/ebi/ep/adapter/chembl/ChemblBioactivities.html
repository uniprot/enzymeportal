3   import java.util.ArrayList;
4   import java.util.Collection;
5   import java.util.HashMap;
6   import java.util.Map;

8   /**
9    * Bioactivities for a given ChEMBL ID (may be target or compound).
10   * @author rafa
11   * @since 1.0.0
12   */
13  public class ChemblBioactivities {
14  
15      /**
16       * Map of ChEMBL IDs to their bioactivities. If this object is for a
17       * target/compound then the keys of this map are compound/target IDs
18       * respectively.
19       */
20      private Map<String, ChemblBioactivity> m =
21              new HashMap<String, ChemblBioactivity>();
22  
23      public void addBioactivity(String chemblId, int confidence,
24              String assayType){
25          if (!m.containsKey(chemblId)){
26              m.put(chemblId, new ChemblBioactivity());
27          }
28          m.get(chemblId).addAssay();
29          if (confidence > 4) m.get(chemblId).addConf4();
30          if (confidence == 9) m.get(chemblId).addConf9();
31          if ("F".equals(assayType)) m.get(chemblId).addFunct();
32      }
33  
34      /**
35       * Retrieves the map of ChEMBL IDs to their bioactivities
36       * @return
37       */
38      public Map<String, ChemblBioactivity> getMap() {
39          return m;
40      }
41  
42      /**
43       * Filters the bioactivities in order to keep only those matching minimum
44       * requirements.
45       * @param minAssays minimum number of assays performed.
46       * @param minConf4 minimum fraction of assays with confidence 4 or higher.
47       * @param minConf9 minimum fraction of assays with confidence 9.
48       * @param minFunc minimum fraction of functional assays.
49       * @return a collection of ChEMBL IDs which passed the filter.
50       */
51      public Collection<String> filter(int minAssays, double minConf4,
52              double minConf9, double minFunc){
53          Collection<String> filtered = new ArrayList<String>();
54          for (String chemblId : m.keySet()) {
55              ChemblBioactivity bioact = m.get(chemblId);
56              if (bioact.getAssays() >= minAssays
57                      && bioact.getConf4() >= minConf4
58                      && bioact.getConf9() >= minConf9
59                      && bioact.getFunct() >= minFunc){
60                  filtered.add(chemblId);
61              }
62          }
63          return filtered;
64      }
65  
66      @Override
67      public String toString() {
68          StringBuilder sb = new StringBuilder("[BIOACTIVITIES]\n");
69          for (String chemblId : m.keySet()) {
70              sb.append('\t').append(chemblId).append(": ")
71                      .append(m.get(chemblId)).append('\n');
72          }
73          return sb.toString();
74      }
