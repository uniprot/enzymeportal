26  /**
27   * Parser for the <code>chembl-target_component.xml</code> file (ebinocle
28   * schema). It reads the UniProt accession of every entry and any target IDs.
29   * If the accession is an enzyme (that is, already existing in the mega-map)
30   * then the target IDs are used with the chembl-adapter to query for
31   * bioactivities, of which only the most significant will be kept and their
32   * ChEMBL compound IDs used for the cross-references.
33   * @author rafa
34   * @since 1.0.22
35   */

77      @Override
78      public void startElement(String uri, String localName, String qName,
79              Attributes attributes) throws SAXException {
80          currentContext.push(localName);
81          // Update flags:
82          String currentXpath = getCurrentXpath();
83          isEntry = DATABASE_ENTRIES_ENTRY.equals(currentXpath);
84          isRef = DATABASE_ENTRIES_ENTRY_XREFS_REF.equals(currentXpath);
85          isField = DATABASE_ENTRIES_ENTRY_FIELD.equals(currentXpath);
86          
87          if (isField){
88              String name = attributes.getValue("", "name");
89              isAccession = name != null && name.equals("accession");
90          } else if (isRef){
91              String dbName = attributes.getValue("", "dbname");
92              if (dbName != null && dbName.equals("ChEMBL-Target")){
93                  targetIds.add(attributes.getValue("", "dbkey"));
94              }
95          }
96          // Clear placeholder:
97          currentChars.delete(0, Integer.MAX_VALUE);
98      }

100     @Override
101     public void endElement(String uri, String localName, String qName)
102     throws SAXException {
103         if (isAccession){
104             accession = currentChars.toString();
105         } else if (isEntry){
106             if (accession != null && !targetIds.isEmpty()){
107                 entry = mm.getEntryForAccession(MmDatabase.UniProt, accession);
108                 if (entry != null){
109                     LOGGER.info(accession + " is enzyme.");
110                     Collection<XRef> xrefs = getChemblXrefs();
111                     try {
112                         if (xrefs != null && !xrefs.isEmpty()){
113                             mm.write(Collections.singleton(entry), xrefs);
114                         }
115                     } catch (IOException e) {
116                         throw new SAXException("Adding entry to mega-map", e);
117                     } finally {
118                         entry = null;
119                         targetIds.clear();
120                     }
121                 }
122             }
123             accession = null;
124             targetIds.clear();
125         }
126         currentContext.pop();
127         // Update flags:
128         String currentXpath = getCurrentXpath();
129         isEntry = DATABASE_ENTRIES_ENTRY.equals(currentXpath);
130         isRef = false;
131         isField = false;
132         isAccession = false;
133     }

142     /**
143      * Adds cross references to any ChEMBL compound IDs which show any
144      * bioactivity against the <code>{@link #targetIds}</code> of the element
145      * being processed.
146      * @return a collection of cross-references from UniProt to ChEMBL, or
147      *      <code>null</code> if none found.
148      */
149     private Collection<XRef> getChemblXrefs() {
150         Collection<XRef> xrefs = null;
151         try {
152             final Collection<String> chemblCompoundIds =
153                     getFilteredChemblCompounds();
154             if (chemblCompoundIds != null){
155                 xrefs = new HashSet<XRef>();
156                 for (String chemblCompoundId : chemblCompoundIds) {
157                     Entry chemblEntry = new Entry();
158                     chemblEntry.setDbName(MmDatabase.ChEMBL.name());
159                     chemblEntry.setEntryId(chemblCompoundId);
160                     chemblEntry.setEntryName(chemblAdapter
161                             .getPreferredName(chemblCompoundId));
162                     XRef xref = new XRef();
163                     xref.setFromEntry(entry);
164                     xref.setRelationship(Relationship.between(
165                             MmDatabase.UniProt, MmDatabase.ChEMBL)
166                             .name());
167                     xref.setToEntry(chemblEntry);
168                     xrefs .add(xref);
169                 }
170             }
171         } catch (Exception e){
172             LOGGER.error(targetIds, e);
173         }
174         return xrefs;
175     }

177     /**
178      * Gets bioactivities from WS for the target IDs stored in the field
179      * <code>{@link #targetIds}</code> and filters them.
180      * @return a collection of ChEMBL compound IDs which have significant
181      *      bioactivities, or <code>null</code> if no bioactivities are found
182      *      for the target ID.
183      * @throws ChemblAdapterException
184      */
185     private Collection<String> getFilteredChemblCompounds()
186     throws ChemblAdapterException {
187         Collection<String> filteredBioactivities = null;
188         for (String targetId : targetIds) {
189             ChemblBioactivities bioactivities =
190                     chemblAdapter.getTargetBioactivities(targetId);
191             if (bioactivities != null){
192                 filteredBioactivities = bioactivities.filter(
193                         chemblAdapter.getConfig().getMinAssays(),
194                         chemblAdapter.getConfig().getMinConf4(),
195                         chemblAdapter.getConfig().getMinConf9(),
196                         chemblAdapter.getConfig().getMinFunc());
197                 LOGGER.debug(bioactivities.getMap().size()
198                         + " bioactivities down to "
199                         + filteredBioactivities.size());
200             }
201         }
202         return filteredBioactivities;
203     }
