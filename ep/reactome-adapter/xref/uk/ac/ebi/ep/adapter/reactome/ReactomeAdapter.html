69      public EnzymeModel addReactionDescriptions(EnzymeModel enzymeModel)
70  	throws ReactomeServiceException {
71          List<ReactionPathway> reactionPathways = enzymeModel.getReactionpathway();
72          //get the reaction and pathways with rhea ws results set
73          for (ReactionPathway reactionPathway: reactionPathways) {
74              EnzymeReaction enzymeReaction = reactionPathway.getReaction();
75             // for (Pathway pathway: pathways) {
76              if (enzymeReaction != null) {
77                  //get Reactome reaction id retrieved from Rhea
78                  List<Object> reactionLinks = enzymeReaction.getXrefs();
79                  String reactomeUrl = null;
80                  if (reactionLinks.size() > 0) {
81                  	for (Object reactionLink : reactionLinks) {
82  						reactomeUrl = (String) reactionLink;
83  						if (!reactomeUrl.contains("REACT_")) continue;
84  						// Only Reactome xrefs, as descriptions come from there
85  	                    try {
86  	                        final String[] split = reactomeUrl.split("=");
87  	                        // Some xrefs come as whole URLs, others as plain IDs
88  							String reactomeAccession = split.length > 1?
89  									split[1] : split[0];
90  	                        String reactionDesc = getReactionDescription(reactomeAccession);
91  	                        enzymeReaction.setDescription(reactionDesc);
92  	                        // XXX: If more than one xref to Reactome, only first description is taken
93  	                        break;
94  	                    }
95  	                    catch (ReactomeServiceException ex) {
96  	                        throw new ReactomeServiceException("Failed to retrieve reaction desciption for " +reactomeUrl, ex);
97  	                    }
98  					}
99                  }
100             }
101         }
102         return enzymeModel;
103     }

105     public List<Pathway> getPathways(List<String> stableIds)
106 	throws ReactomeServiceException {
107         ExecutorService pool = Executors.newCachedThreadPool();
108 	    CompletionService<Pathway> ecs =
109 	    		new ExecutorCompletionService<Pathway>(pool);
110         List<Pathway> pathways = new ArrayList<Pathway>();
111         try {
112         	logger.debug(" -RP- before sending jobs");
113 	        for (String id: stableIds) {
114 	            GetPathwayCaller callable = new ReactomeCallable.GetPathwayCaller(id, config);
115 	            ecs.submit(callable);
116 	        }
117         	logger.debug(" -RP- before retrieving jobs");
118 	        for (int i = 0; i < stableIds.size(); i++) {
119 				try {
120 					Future<Pathway> future =
121 							ecs.poll(config.getTimeout(), TimeUnit.MILLISECONDS);
122 	    			if (future != null){
123 		                pathways.add(future.get());
124 	    			} else {
125 	    				logger.warn("Job did not return");
126 	    			}
127 	            } catch (InterruptedException ex) {
128 	                throw new ReactomeServiceException("The thread was interrupted while retieving from Reactome pathway " +stableIds.get(i), ex);
129 	            } catch (ExecutionException ex) {
130 	                throw new ReactomeServiceException("Failed to execute the thread to retrieve from Reactome pathway " +stableIds.get(i), ex);
131 	            }
132 			}
133         } finally {
134             pool.shutdown();
135         }
136     	logger.debug(" -RP- before returning");
137         return pathways;
138     }
