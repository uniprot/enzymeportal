41      public static List<Disease> getDiseases(List<Comment> commentList) {
42          List<Disease> diseases = new ArrayList<Disease>();
43          for (Comment comment : commentList) {
44              Disease disease = new Disease();
45              DiseaseCommentImpl castedComment = (DiseaseCommentImpl) comment;
46              //Work around to delete Note= from the comment
47              String commentText = castedComment.getValue().replace("Note=", "");
48              String id = String.valueOf(castedComment.getId());
49              disease.setId(id);
50              disease.setDescription(commentText);
51              diseases.add(disease);
52          }
53          return diseases;
54      }
