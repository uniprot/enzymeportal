13  /**
14   * Comparison for lists of items. It does not take into account their order, so
15   * for this class <code>[ "1", "2" ]</code> is not different from
16   * <code>[ "2", "1" ]</code>.<br>
17   * This comparison requires that the items class implements properly the equals
18   * and hashCode methods.
19   * 
20   * @author rafa
21   * @since 1.1.0
22   */
23  public class ListComparison extends AbstractComparison<List<?>> {

25      public ListComparison(final List<?> l1, final List<?> l2) {
26          compared = new List<?>[] { l1, l2 };
27          init(l1, l2);
28      }

30      @Override
31      protected void getSubComparisons(List<?> l1, List<?> l2) {
32          if (l1 == null) l1 = new ArrayList<Object>();
33          if (l2 == null) l2 = new ArrayList<Object>();
34          List<?> common = ListUtils.intersection(l1, l2);
35          List<?> onlyL1 = ListUtils.subtract(l1, common);
36          List<?> onlyL2 = ListUtils.subtract(l2, common);
37          int c = 0;
38          for (Object o : common) {
39              // We do not use just Object o, they are different objects!
40              subComparisons.put("same-" + c++, getItemComparison(
41                      l1.get(l1.indexOf(o)), l2.get(l2.indexOf(o))));
42          }
43          final int min = Math.min(onlyL1.size(), onlyL2.size());
44          int i = 0;
45          for (; i < min; i++) {
46              subComparisons.put("diff-" + i,
47                      getItemComparison(onlyL1.get(i), onlyL2.get(i)));
48          }
49          if (onlyL1.size() > onlyL2.size()){
50              for (; i < onlyL1.size(); i++){
51                  subComparisons.put("diff-" + i,
52                          getItemComparison(onlyL1.get(i), null));
53              }
54          } else if (onlyL1.size() < onlyL2.size()){
55              for (; i < onlyL2.size(); i++){
56                  subComparisons.put("diff-" + i,
57                          getItemComparison(null, onlyL2.get(i)));
58              }
59          }
60          differ = !onlyL1.isEmpty() || !onlyL2.isEmpty();
61      }

63      /**
64       * Factory method to get a comparison for two objects according to their
65       * class.
66       * 
67       * @param o1
68       * @param o2
69       * @return an adequate Comparison.
70       */
71      private Comparison<?> getItemComparison(Object o1, Object o2) {
72          Comparison<?> itemComparison = null;
73          Class<? extends Object> theClass = o1 != null?
74                  o1.getClass() : o2.getClass();
75          if (theClass.equals(String.class)) {
76              itemComparison = new StringComparison((String) o1, (String) o2);
77          } else if (theClass.equals(ReactionPathway.class)) {
78              itemComparison = new ReactionPathwayComparison(
79                      (ReactionPathway) o1, (ReactionPathway) o2);
80          } else if (theClass.equals(Pathway.class)) {
81              itemComparison = new PathwayComparison((Pathway) o1, (Pathway) o2);
82          } else if (theClass.equals(Molecule.class)) {
83              itemComparison = new MoleculeComparison(
84                      (Molecule) o1, (Molecule) o2);
85          } else if (theClass.equals(Disease.class)) {
86              itemComparison = new DiseaseComparison((Disease) o1, (Disease) o2);
87          }
88          return itemComparison;
89      }

92      public String toString() {
93          return subComparisons == null || subComparisons.isEmpty()? "" :
94                  subComparisons.entrySet().iterator().next().getValue()
95                  .toString();
